<ThemeProvider>
  <DataGrid
    model={[
      {
        property: 'name',
        displayName: 'Name',
      },
      {
        property: 'roles',
        displayName: 'User groups',
        Component: entity =>
          entity.roles.map((r, i, { length }) => (
            <div>
              <i>{r}</i>
              {i === length - 1 ? '' : ', '}
            </div>
          )),
      },
    ]}
    data={[{ name: 'John', roles: ['admin', 'editor', 'writer'] }]}
  />
</ThemeProvider>
